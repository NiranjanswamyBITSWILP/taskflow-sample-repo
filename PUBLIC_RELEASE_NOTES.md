```bash
# Docker (Recommended)
docker-compose up -d
# Access: http://localhost:3000

# Local Development
cd backend && npm install && npm start
cd frontend && npm install && npm start  # new terminal
docker run -d -p 27017:27017 mongo:7.0  # new terminal
```
